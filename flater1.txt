import 'package:flutter/material.dart';

void main() {
  runApp(const PhishingApp());
}

class PhishingApp extends StatelessWidget {
  const PhishingApp({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      title: 'Phishing Detector - Single Page',
      theme: ThemeData(primarySwatch: Colors.blue),
      home: const SinglePageHome(),
      debugShowCheckedModeBanner: false,
    );
  }
}

/* ===========================
   SinglePageHome: صفحة واحدة تحتوي
   - إدخال الرابط
   - منطقة عرض النتائج (تتغير حسب الحالة)
   =========================== */
class SinglePageHome extends StatefulWidget {
  const SinglePageHome({super.key});

  @override
  State<SinglePageHome> createState() => _SinglePageHomeState();
}

class _SinglePageHomeState extends State<SinglePageHome> {
  final TextEditingController _controller = TextEditingController();
  AnalysisResult? _lastResult;
  String? _lastUrl;

  void _runAnalysis() {
    final raw = _controller.text.trim();
    if (raw.isEmpty) {
      // حالة الحقل الفارغ
      setState(() {
        _lastResult = null;
        _lastUrl = null;
      });
      _showEmptyDialog();
      return;
    }

    final url = _normalizeUrl(raw);
    final analysis = LinkAnalyzer.analyze(url);

    setState(() {
      _lastUrl = url;
      _lastResult = analysis;
    });
  }

  String _normalizeUrl(String input) {
    if (!input.startsWith('http://') && !input.startsWith('https://')) {
      return 'http://$input';
    }
    return input;
  }

  void _resetAll() {
    _controller.clear();
    setState(() {
      _lastResult = null;
      _lastUrl = null;
    });
  }

  void _showEmptyDialog() {
    showDialog(
      context: context,
      builder: (_) => AlertDialog(
        title: const Text("حقل فارغ"),
        content: const Text("الرجاء إدخال رابط صحيح قبل الفحص."),
        actions: [
          TextButton(onPressed: () => Navigator.pop(context), child: const Text("حسناً")),
        ],
      ),
    );
  }

  Widget _buildInputCard() {
    return Card(
      elevation: 4,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
      child: Padding(
        padding: const EdgeInsets.all(14.0),
        child: Column(
          children: [
            const Text("أدخل الرابط للتحقق:", style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold)),
            const SizedBox(height: 10),
            TextField(
              controller: _controller,
              decoration: const InputDecoration(
                border: OutlineInputBorder(),
                hintText: "مثال: https://example.com أو google.com",
              ),
              keyboardType: TextInputType.url,
            ),
            const SizedBox(height: 12),
            Row(
              children: [
                Expanded(
                  child: ElevatedButton.icon(
                    onPressed: _runAnalysis,
                    icon: const Icon(Icons.search),
                    label: const Text("تحقق الآن"),
                  ),
                ),
                const SizedBox(width: 10),
                OutlinedButton(
                  onPressed: _resetAll,
                  child: const Text("مسح"),
                )
              ],
            ),
          ],
        ),
      ),
    );
  }

  Widget _buildResultArea() {
    if (_lastUrl == null || _lastResult == null) {
      return Card(
        elevation: 2,
        margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
        child: Padding(
          padding: const EdgeInsets.all(14.0),
          child: Column(
            children: const [
              Icon(Icons.info_outline, size: 64, color: Colors.blue),
              SizedBox(height: 8),
              Text(
                "لم يتم فحص أي رابط بعد.\nأدخل رابط واضغط (تحقق الآن) لعرض النتيجة هنا.",
                textAlign: TextAlign.center,
                style: TextStyle(fontSize: 16),
              )
            ],
          ),
        ),
      );
    }

    switch (_lastResult!.type) {
      case ResultType.safe:
        return _safeWidget(_lastUrl!, _lastResult!);
      case ResultType.suspicious:
        return _suspiciousWidget(_lastUrl!, _lastResult!);
      case ResultType.phishy:
        return _phishyWidget(_lastUrl!, _lastResult!);
      default:
        return _genericWidget(_lastUrl!, _lastResult!);
    }
  }

  Widget _safeWidget(String url, AnalysisResult details) {
    return Card(
      color: Colors.green[50],
      elevation: 3,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
      child: Padding(
        padding: const EdgeInsets.all(14.0),
        child: Column(
          children: [
            const Icon(Icons.check_circle_outline, size: 72, color: Colors.green),
            const SizedBox(height: 8),
            Text("الرابط يبدو آمنًا مبدئياً", style: TextStyle(fontSize: 18, color: Colors.green[900])),
            const SizedBox(height: 8),
            SelectableText("الرابط: $url", style: const TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 12),
            const Text("تفاصيل الفحص:", style: TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 6),
            ...details.reasons.map((r) => ListTile(leading: const Icon(Icons.info), title: Text(r))),
            const SizedBox(height: 6),
            const Text("نصيحة: تأكد دائماً من الشهادة وإدخال البيانات الحساسة فقط في المواقع الموثوقة."),
          ],
        ),
      ),
    );
  }

  Widget _suspiciousWidget(String url, AnalysisResult details) {
    return Card(
      color: Colors.orange[50],
      elevation: 3,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
      child: Padding(
        padding: const EdgeInsets.all(14.0),
        child: Column(
          children: [
            const Icon(Icons.warning_amber_outlined, size: 72, color: Colors.orange),
            const SizedBox(height: 8),
            Text("الرابط مريب — يُنصح بالحذر", style: TextStyle(fontSize: 18, color: Colors.orange[900])),
            const SizedBox(height: 8),
            SelectableText("الرابط: $url", style: const TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 12),
            const Text("الأسباب المكتشفة:", style: TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 6),
            ...details.reasons.map((r) => Card(child: Padding(padding: const EdgeInsets.all(8.0), child: Text(r)))),
            const SizedBox(height: 10),
            const Text("إجراءات مقترحة:\n• لا تدخل معلومات حساسة.\n• راجع الدومين جيدًا وتحقق من الأخطاء الإملائية."),
          ],
        ),
      ),
    );
  }

  Widget _phishyWidget(String url, AnalysisResult details) {
    return Card(
      color: Colors.red[50],
      elevation: 3,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
      child: Padding(
        padding: const EdgeInsets.all(14.0),
        child: Column(
          children: [
            const Icon(Icons.dangerous_outlined, size: 72, color: Colors.red),
            const SizedBox(height: 8),
            Text("الرابط مشبوه جداً — لا تصل إليه", style: TextStyle(fontSize: 18, color: Colors.red[900])),
            const SizedBox(height: 8),
            SelectableText("الرابط: $url", style: const TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 12),
            const Text("الأسباب:", style: TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 6),
            ...details.reasons.map((r) => Card(child: Padding(padding: const EdgeInsets.all(8.0), child: Text(r)))),
            const SizedBox(height: 10),
            const Text("توصيات أمنية:\n• لا تدخل أي بيانات.\n• احذف المصدر الذي وصلتك منه الرسالة.\n• افحص جهازك إن لزم."),
          ],
        ),
      ),
    );
  }

  Widget _genericWidget(String url, AnalysisResult details) {
    return Card(
      elevation: 2,
      margin: const EdgeInsets.symmetric(horizontal: 16, vertical: 12),
      child: Padding(
        padding: const EdgeInsets.all(14.0),
        child: Column(
          children: [
            const Icon(Icons.help_outline, size: 64),
            const SizedBox(height: 8),
            const Text("نتيجة غير محددة بدقة", style: TextStyle(fontSize: 18)),
            const SizedBox(height: 8),
            SelectableText("الرابط: $url", style: const TextStyle(fontWeight: FontWeight.bold)),
            const SizedBox(height: 8),
            ...details.reasons.map((r) => ListTile(leading: const Icon(Icons.info), title: Text(r))),
          ],
        ),
      ),
    );
  }

  String _typeLabel(ResultType t) {
    switch (t) {
      case ResultType.safe:
        return 'آمن';
      case ResultType.suspicious:
        return 'مريب';
      case ResultType.phishy:
        return 'مشبوه جداً';
      default:
        return 'غير محدد';
    }
  }

  @override
  Widget build(BuildContext context) {
    return Scaffold(
      appBar: AppBar(
        title: const Text("كشف الاحتيال - صفحة واحدة"),
      ),
      body: SafeArea(
        child: Column(
          children: [
            _buildInputCard(),
            Expanded(
              child: SingleChildScrollView(
                child: _buildResultArea(),
              ),
            ),
            Container(
              padding: const EdgeInsets.symmetric(vertical: 10, horizontal: 16),
              child: Row(
                children: [
                  if (_lastResult != null)
                    Row(
                      children: [
                        Text("الدرجة: ${_lastResult!.score}"),
                        const SizedBox(width: 10),
                        Text("التصنيف: ${_typeLabel(_lastResult!.type)}", style: const TextStyle(fontWeight: FontWeight.bold)),
                      ],
                    )
                  else
                    const Text("لم يتم الفحص بعد"),
                  const Spacer(),
                  TextButton(onPressed: _resetAll, child: const Text("إعادة ضبط")),
                ],
              ),
            ),
          ],
        ),
      ),
    );
  }
}

/*=======================
   نموذج نتيجة التحليل (Data class)
   =========================== */
enum ResultType { safe, suspicious, phishy, unknown }

class AnalysisResult {
  final ResultType type; ====
  final List<String> reasons;
  final bool hasSsl;
  final bool containsIp;
  final bool looksLikeFamous;
  final bool hasDangerKeywords;
  final int score;

  AnalysisResult({
    required this.type,
    required this.reasons,
    required this.hasSsl,
    required this.containsIp,
    required this.looksLikeFamous,
    required this.hasDangerKeywords,
    required this.score,
  });
}

/* ===========================
   LinkAnalyzer: قواعد فحص نصية (heuristics)
   يعمل بدون إنترنت
   =========================== */
class LinkAnalyzer {
  static AnalysisResult analyze(String url) {
    final lower = url.toLowerCase();

    // SSL
    bool hasSsl = lower.startsWith('https://');

    // IP في الدومين؟
    RegExp ipRegex = RegExp(r'^(http[s]?:\/\/)?(\d{1,3}\.){3}\d{1,3}');
    bool containsIp = ipRegex.hasMatch(lower);

    // كلمات خطيرة
    List<String> danger = ['login', 'verify', 'update', 'bank', 'confirm', 'secure', 'signin'];
    bool hasDangerKeywords = danger.any((w) => lower.contains(w));

    // يشبه موقع مشهور؟ (خدع هجائية)
    List<String> famousRoots = ['google', 'facebook', 'youtube', 'amazon', 'paypal'];
    bool looksLikeFamous = false;
    for (var f in famousRoots) {
      if (lower.contains('$f.') || lower.contains('.$f')) {
        // وجود الدومين بشكل صحيح => ليس بالضرورة مشبوه
      } else if (lower.contains(f) && !lower.contains('$f.')) {
        looksLikeFamous = true;
        break;
      }
    }

    // طول الرابط
    bool veryLong = url.length > 80;

    // أسباب
    List<String> reasons = [];
    if (!hasSsl) reasons.add("لا يوجد HTTPS / شهادة SSL.");
    if (containsIp) reasons.add("يحتوي على عنوان IP بدلًا من دومين.");
    if (hasDangerKeywords) reasons.add("يحتوي كلمات حسّاسة مثل login/verify/bank.");
    if (looksLikeFamous) reasons.add("يشبه اسم موقع مشهور (محاولة تقليد).");
    if (veryLong) reasons.add("طول الرابط كبير جداً — قد يُخفي مسار خبيث.");
    if (reasons.isEmpty) reasons.add("لم تُكتشف علامات مشبوهة واضحة.");

    // حساب درجة الشبهة
    int score = 0;
    if (!hasSsl) score += 2;
    if (containsIp) score += 3;
    if (hasDangerKeywords) score += 2;
    if (looksLikeFamous) score += 3;
    if (veryLong) score += 1;

    ResultType finalType;
    if (score >= 6) {
      finalType = ResultType.phishy;
    } else if (score >= 3) {
      finalType = ResultType.suspicious;
    } else if (score <= 2) {
      finalType = ResultType.safe;
    } else {
      finalType = ResultType.unknown;
    }

    return AnalysisResult(
      type: finalType,
      reasons: reasons,
      hasSsl: hasSsl,
      containsIp: containsIp,
      looksLikeFamous: looksLikeFamous,
      hasDangerKeywords: hasDangerKeywords,
      score: score,
    );
  }
}
